# Composant `<LatestBlogPosts />`

Ce composant React permet d’afficher une liste d’articles du blog sous forme de cartes responsives, avec de nombreux paramètres de filtrage.

---

## Utilisation de base

```jsx
<LatestBlogPosts count={3} />
```

## Exemple avec un auteur spécifique

```jsx
<LatestBlogPosts author="docux" count={5} />
```

## Exemple avec plusieurs auteurs

```jsx
<LatestBlogPosts author={["docux", "kiki"]} count={10} description={true} />
```

## Exemple avec un tag spécifique

```jsx
<LatestBlogPosts tag="actu" count={6} />
```

## Exemple avec la description

```jsx
<LatestBlogPosts description={true} />
```
```jsx
export default function LatestBlogPosts({
  tag,                // Tag à filtrer (optionnel)
  author,             // Auteur(s) à filtrer (string, array ou undefined)
  count = 3,          // Nombre d'articles à afficher (par défaut 3)
  description = false // Afficher la description en haut de la carte (optionnel)
}) {
  // Récupération de tous les articles du blog
  let filtered = posts;

  // Filtrage par tag si la prop 'tag' est fournie
  if (tag) {
    filtered = filtered.filter((post) =>
      post.tags.some((t) => t === tag || (t.label && t.label === tag))
    );
  }

  // Filtrage par auteur(s) si la prop 'author' est fournie et non vide
  if (author && author.length > 0) {
    // On accepte un string ou un tableau d'auteurs
    const authorList = Array.isArray(author) ? author : [author];
    filtered = filtered.filter((post) =>
      Array.isArray(post.authors)
        ? post.authors.some((a) => authorList.includes(a))
        : authorList.includes(post.authors)
    );
  }

  // On limite le nombre d'articles affichés à 'count'
  const latest = filtered.slice(0, count);

  // Si aucun article trouvé, on affiche un message
  if (!latest.length) return <p>Aucun article trouvé.</p>;

  // Affichage des cartes d'articles dans une grille responsive Infima
  return (
    <div className="row">
      {latest.map((post) => (
        <div className="col col--4" key={post.permalink} style={{ marginBottom: '2rem', display: 'flex' }}>
          <div className="card" style={{ width: '100%', display: 'flex', flexDirection: 'column', height: '100%' }}>
            {/* Affichage de l'image de l'article si présente */}
            <div className="card__image">
              {post.image && (
                <img
                  src={useBaseUrl(post.image)}
                  alt={post.title}
                  style={{
                    width: '100%',
                    height: 180,
                    objectFit: 'cover',
                    borderTopLeftRadius: 8,
                    borderTopRightRadius: 8,
                  }}
                />
              )}
            </div>
            {/* Corps de la carte : titre, description, auteur, date */}
            <div className="card__body" style={{ flex: 1 }}>
              <h3>{post.title}</h3>
              {/* Affiche la description en haut si demandé */}
              {description && post.description && (
                <div style={{ color: '#6c63ff', fontWeight: 'bold', marginBottom: 6 }}>
                  {post.description}
                </div>
              )}
              {/* Affiche l'auteur et la date */}
              <p style={{ color: '#888', fontSize: '0.95em', marginBottom: 8 }}>
                {post.authors && post.authors.length > 0 && (
                  <>Par <b>{Array.isArray(post.authors) ? post.authors.join(', ') : post.authors}</b> – </>
                )}
                {post.date && (
                  <span>{new Date(post.date).toLocaleDateString()}</span>
                )}
              </p>
            </div>
            {/* Pied de carte : bouton "Lire plus" */}
            <div className="card__footer" style={{ textAlign: 'right' }}>
              <Link className="button button--primary button--sm" to={post.permalink}>
                Lire plus
              </Link>
            </div>
          </div>
        </div>
      ))}
    </div>
  );
}
```
